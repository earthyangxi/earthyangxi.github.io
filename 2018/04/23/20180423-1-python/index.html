import

# 直接导入
import math
print(math.cos(math.pi))

# 导入函数和变量
from math import cos, pi
print(cos(pi))

# 起别名
import math as m
print(m.cos(m.pi))

# 导入所有
from math import *
print(cos(pi))

# 单下划线开头表示protected
# 双下划线开头表示private
class A:
    """Class A""""
    def __init__(self, x, y, name):
        self.x = x
        self.y = y
        self._name = name

    def introduce(self):
        print(self.name)

map, reduce, filter

map(lambda x, y: x + y, [1, 2, 3], [5, 6, 7]) # [6, 8, 10]
reduce(lambda x, y: x +y, [1, 2, 3, 4]) # ((1+2)+3)+4=10
filter(lambda x: x % 2, [1, 2, 3, 4, 5]) # [1, 3, 5]
